```html
<!DOCTYPE html> 					<!-- html5 버전으로 자동으로 실행 -->

<html>							<!-- html 의 시작 -->
	<head>							<!-- head 의시작 -->
		<title> My First Website </title>			<!-- 창의 제목이 됨 -->
		<meta charset="utf-8"> 					<!-- 한국어 지원 --> 
		<style>								
			h1 {
			  color: lime;
			}
			h2 {
			  font-size: 50px;
			}
			h3 {
			  text-align: center;
			}
			h4 {
			  margin-left: 50px;
			}
		</style>
	</head>							<!-- head 끝 -->
	<body>							<!-- body 의시작 -->
		<h1> Biggest letter </h1> 				<!-- 글자크기 조절(h1~h5 큰순서대로 ) -->
		<h2> 2nd </h2>
		<h3> 3rd </h3>
		<h4> 4rd </h4>
		<h5> 5rd </h5>
		<a href="http://google.com"
		target="_blank">구글로 가는 링크</a>			 <!-- a 태그는 하이퍼 링크, target="_blank" 는 새창에서 열리게 하기 -->
		<p>							<!-- 문단만들기 -->
			<b>단축키 정리</b><br>
			ctrl+} = 전체 들여쓰기<br>
		</p>
	</body>							<!-- body 끝 -->
</html>							<!-- html 의 끝 -->

```
